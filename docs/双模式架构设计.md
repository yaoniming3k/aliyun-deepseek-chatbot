            'max_tokens' => intval(get_option('aliyun_max_tokens', 2000))
        ];

            return null;
        }

        $json = trim(substr($line, 6));
        if ($json === '[DONE]') {
            return ['done' => true];
        }

        $data = json_decode($json, true);
        return [
            'content' => $data['choices'][0]['delta']['content'] ?? '',
            'done' => false
        ];

        $messages = $this->get_conversation_history($session_id);

    }

    private function save_conversation_history($session_id, $messages) {
        global $wpdb;
        $table_name = $wpdb->prefix . 'aliyun_chat_history';

        $wpdb->replace(
            $table_name,
            [
                'session_id' => $session_id,
                'messages' => json_encode($messages, JSON_UNESCAPED_UNICODE),
                'updated_at' => current_time('mysql')
